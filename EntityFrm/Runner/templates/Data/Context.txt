using System;
using System.Linq;
using System.Threading.Tasks;
using Microsoft.EntityFrameworkCore;
using {SOLUCAO}.{PROJETO}.Domain;
using {SOLUCAO}.Core.Data;
using {SOLUCAO}.Core.Messages;

namespace {SOLUCAO}.{PROJETO}.Data
{
    public class {PROJETO}Context : DbContext, IUnitOfWork
    {
        public {PROJETO}Context(DbContextOptions<{PROJETO}Context> options)
            : base(options) { }

[Context_SET]

        public async Task<bool> Commit()
        {
            foreach (var entry in ChangeTracker.Entries().Where(entry => entry.Entity.GetType().GetProperty("DataCadastro") != null))
            {
                if (entry.State == EntityState.Added)
                {
                    entry.Property("DataCadastro").CurrentValue = DateTime.Now;
                }

                if (entry.State == EntityState.Modified)
                {
                    entry.Property("DataCadastro").IsModified = false;
                }
            }
            
            return await base.SaveChangesAsync() > 0;
        }
    }
}
